                $path = Join-Path $scriptRoot "Reports\benchmark_$(Get-Date -Format 'yyyyMMdd_HHmmss').txt"

                Update-Status -Text "SFC scan complete" -State "Success"
            } catch { [System.Windows.MessageBox]::Show("Error: $($_.Exception.Message)", "Error", "OK", "Error") }
        })
    }
    
    $btn = Find-UIElement "RunDISMCheckBtn"
    if ($btn) {
        $btn.Add_Click({
            try {
                Update-Status -Text "Running DISM CheckHealth..." -State "Working"

                Update-Status -Text "Running full system repair..." -State "Working"

} catch {
