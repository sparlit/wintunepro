$ErrorActionPreference = "Continue"
$scriptRoot = 'E:\WinTunePro'
$allErrors = @()
$pass = 0; $fail = 0

function Test-Func($name, $code) {
    try { & $code | Out-Null; $script:pass++; Write-Host "  [OK] $name" -ForegroundColor Green } catch { $script:fail++; $script:allErrors += "$name : $($_.Exception.Message)"; Write-Host "  [ERR] $name : $($_.Exception.Message)" -ForegroundColor Red }
}

Write-Host "=== FULL TEST ===" -ForegroundColor Cyan

# Load core
$core = @("Config.ps1","Logger.ps1","State.ps1","AppCore.ps1","Helpers.ps1","SafetyNet.ps1","Safety.ps1","Rollback.ps1","Resume.ps1","HealthScore.ps1","ReportGenerator.ps1","AutoElevate.ps1","ParallelEngine.ps1","SelfHealing.ps1","ToolManager.ps1","Watchdog.ps1","ProfileManager.ps1","AutomationEngine.ps1")
foreach ($f in $core) { try { . "$scriptRoot\Core\$f" } catch { $allErrors += "Core\$f : $($_.Exception.Message)"; $fail++ } }
try { Initialize-State -AppRoot $scriptRoot; Initialize-ConfigPaths -RootPath $scriptRoot; Initialize-Logger -LogDirectory (Join-Path $scriptRoot "Logs") -SessionIdentifier "T"; Initialize-Application -RootPath $scriptRoot } catch { $allErrors += "Init : $($_.Exception.Message)" }

# Load modules
$mods = Get-ChildItem "$scriptRoot\Modules" -Recurse -Filter '*.ps1' | Where-Object { $_.FullName -notlike '*Sophia*' }
$mOk = 0; $mFail = 0
foreach ($m in $mods) { try { . $m.FullName; $mOk++ } catch { $mFail++; $allErrors += "$($m.Name) : $($_.Exception.Message)" } }

# Load UI
try { . "$scriptRoot\UI\MainWindow.ps1" } catch { $allErrors += "UI : $($_.Exception.Message)" }

# Test all button functions

Write-Host "`n=== RESULTS ===" -ForegroundColor Cyan

Write-Host "Functions: $pass/$($pass+$fail)" -ForegroundColor $(if ($fail -eq 0) { "Green" } else { "Red" })
Write-Host "Total errors: $($allErrors.Count)" -ForegroundColor $(if ($allErrors.Count -eq 0) { "Green" } else { "Red" })

if ($allErrors.Count -gt 0) {
    $allErrors | Out-File "$scriptRoot\errors_found.txt" -Encoding UTF8
    Write-Host "Errors saved to errors_found.txt" -ForegroundColor DarkGray
}
